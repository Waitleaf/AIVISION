CREATE DATABASE AIVISION;

USE AIVISION;

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(255) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL
    passwordKey VARCHAR(255) NOT NULL
    passwordValue VARCHAR(255) NOT NULL
);
